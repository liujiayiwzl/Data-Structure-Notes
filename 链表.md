# 链表

 ```c
 #include <stdio.h>
#include <stdlib.h>

typedef int ElemType;

typedef struct node//一个名为Node的结构体
{
	ElemType data;//存储数据，ElemType（int）类型的数据
	struct node* next;//存储data后面的一个位置的地址，称为后继
}Node;

//1.初始化链表（创造头节点）
Node* initList()
{
	Node* head = (Node*)malloc(sizeof(Node));
	head->data = 0;//创造头节点，为0
	head->next = NULL;
	return head;
}

//2.添加元素（头插法）
int insertHead(Node* L, ElemType e)//传入链表地址，插入参数e
{
	Node* p = (Node*)malloc(sizeof(Node));//创造新节点p
	p->data = e;//新节点数据赋值e
	p->next = L->next;//新节点的后继位置改为原来头节点的后继位置
	L->next = p;//把头节点的后置位置直接改为新的节点整体
	           //保证新节点在头节点后面
	return 1;
}

//3.遍历
void listNode(Node* L)
{
	Node* p = L->next; // 直接从有真实数据的第一个节点开始找
	while (p != NULL)
	{
		printf("%d\n", p->data);
		p = p->next; // 顺藤摸瓜，找下一个人
	}
	printf("\n");
}

//4.尾插法

//获取尾节点
Node* get_tail(Node* L)
{
	Node* p = L;
	while (p->next!=NULL)//遍历，获取到p的next为NULL时
	{
		p = p->next;
	}
	return p;
}

//插入元素
Node* insertTail(Node* tail, ElemType e)
{
	Node* p = (Node*)malloc(sizeof(Node));
	p->data = e;//新节点数据赋值e
	tail->next = p;//尾节点的next赋值给p，p就是新的尾节点
	p->next = NULL;//p的next为NULL
	return p;//返回最新的尾节点
}


//5.指定位置插入
int insertNode(Node* L, int pos, ElemType e)//L头节点，pos插入位置，e插入数据
{
    //用来保存插入位置的前驱节点
	Node* p = L;//p初始化为头节点
	int i = 0;

	if (pos < 1) 
	{
		printf("位置不合法\n");
		return 0;
	}
	//遍历链表找到插入位置的前驱节点
	
	//p停留在目标位置前一个节点
	while (i < pos - 1)
	{
		p = p->next;
		i++;
		if (p == NULL)
		{
			return 0;
		}
	}

	//要插入的节点
	Node* q = (Node*)malloc(sizeof(Node));
	q->data = e;
	q->next = p->next;//将前驱节点的next（下一个节点整体）赋值给新节点的next
	p->next = q;//前驱节点的next改为新节点的整体
	return 1;
}

//6.删除节点
/*
   步骤：
       找到要删除的节点的前置节点p
	   用指针q记录要删除的节点
	   通过改变p的后继节点实现删除
	   释放删除的的节点空间
*/
int deleteNode(Node* L, int pos)
{
	//初始化要删除的节点的前驱
	Node* p = L;
	int i = 0;

	if (pos < 1) 
	{
		printf("位置不合法\n");
		return 0;
	}

	//遍历链表，找到前驱节点
	while (i < pos - 1)
	{
		p = p->next;//p为前驱节点
		i++;
		if (p == NULL)
		{
			return 0;
		}
	}

	if (p->next == NULL)
	{
		printf("要删除的位置错误\n");
		return 0;
	}

	//q指向要删除的节点
	Node* q = p->next;
	p->next = q->next;//前驱节点的next指向删除节点的next（删除后面的节点整体）
	free(q);//释放删除的节点
	return 1;
}


//7.获取链表长度
int listLength(Node* L)
{
	//初始化长度，从链表头节点向后遍历
	Node* p = L;
	int len=0;

	while (p != NULL)
	{
		p = p->next;
		len++;
	}
	return len;
}


//8.释放链表
/*
   步骤：
       指针p指向头节点后第一个节点
	   判断p是否指向NULL
	   不为空，用q记录指针p的后继节点
	   释放指针p指向的节点
	   p 和 q指向同一个节点，循环上面操作
*/
int freeList(Node* L)
{
	//指针p指向头节点后第一个节点
	Node* p=L->next;
	Node* q;

	//判断p是否指向NULL
	//不为空，用q记录指针p的后继节点
	while (p != NULL)
	{
		q = p->next;
		free(p);//释放指针p指向的节点
		p = q;
	}

	//释放完后，头节点的next指向NULL
	L->next = NULL;

	return 1;
}





int main()
{
	//1.初始化链表
	Node* list = initList();//结构体类型的指针，接收初始化的结果
	
	//4.获取尾节点
	Node* tail = get_tail(list);

	tail = insertTail(tail, 30);//插入尾部数据，返回尾节点赋值给tail，方便获取尾节点

	//2.插入元素（头插法）先进的排到后面去
	insertHead(list, 10);
	insertHead(list, 20);

	//3.遍历
	listNode(list);
	
	//5.指定位置插入
	insertNode(list, 2, 40);
	listNode(list);

	//6.删除节点
	deleteNode(list, 2);
	listNode(list);

	//7.链表长度
	printf("长度为%d\n", listLength(list));

	//8.释放链表
	freeList(list);
	printf("长度为%d\n", listLength(list));


	return 0;
}
```
