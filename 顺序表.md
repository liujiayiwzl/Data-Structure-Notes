# 线性表
## 一：线性表的定义与特点
### 1.定义：
- 由n（n>=0）个数据特性相同的元素构成的有限序列。称为线性表

 - <span style="color:orange">线性表是n个数据元素的有限序列，n个数据是相同数据类型<span>

### 2. 特点：
- 线性表元素个数n定义为线性表的长度，当n=0时称为空表。
- 对于非空的线性表或线性结构，其特点是：
- <span style="color:orange">存在唯一的一个被称为“第一个”和“最后一个”的元素

- 除第一个和最后一个元素外，结构中每个数据元素均只有一个<span style="color:orange">前驱和后驱
 

## 二：线性表的顺序表示和实现（顺序表）
### 1. 顺序表定义：
- 用一组连续的内存单元依次存储线性表的各个元素，相邻的元素，实际存储空间也是连续的

### 2.存储结构
```c
#include <stdio.h>
#include <stdlib.h> // 必须引入！因为用到了 malloc 和动态内存分配

#define MAXSIZE 100

// 【考点】解耦思想：给 int 起别名，方便未来一键修改整个表的数据类型
typedef int ElemType; 

/* * 0. 定义顺序表结构体
 * 包含两部分：存放数据的底层仓库（data指针），和当前实际拥有的元素个数（length）
 */
typedef struct
{
    ElemType *data; // 核心：用指针指向堆区开辟的一大块连续内存
    int length;     // 记录当前表里有多少个元素
} SeqList;


/*
 * 1. 顺序表初始化（动态内存分配版）
 * 为什么用 malloc？因为在堆区分配内存更灵活，可以控制生命周期
 */
SeqList* initList()
{
    // 第1步：先给“顺序表结构体”本身申请一个内存位置
    SeqList* L = (SeqList*)malloc(sizeof(SeqList));
    // 第2步：给结构体里的 data 指针申请 100个ElemType大小 的连续空间（底层大仓库）
    L->data = (ElemType*)malloc(sizeof(ElemType) * MAXSIZE); 
    L->length = 0; // 初始长度为0
    
    return L; // 返回这个表的操作钥匙（指针）
}


/*
 * 2. 尾部添加元素（时间复杂度 O(1)）
 */
int appendElem(SeqList* L, ElemType e)
{
    if (L->length >= MAXSIZE) // 防呆设计：满表拦截
    {
        printf("顺序表已满，无法添加\n");
        return 0;
    }
    // L->length 刚好指向当前数组的下一个空位下标
    L->data[L->length] = e; 
    L->length++;
    return 1;
}


/*
 * 3. 遍历输出
 */
void listElem(SeqList* L)
{
    for (int i = 0; i < L->length; i++)
    {
        printf("%d ", L->data[i]);
    }
    printf("\n");
}


/*
 * 4. 插入元素（时间复杂度 O(n)）
 * @param pos: 现实中的位置，从 1 开始算（注意与数组下标 i 差 1）
 */
int insertElem(SeqList* L, int pos, ElemType e)
{
    if (L->length >= MAXSIZE)
    {
        printf("顺序表已满，无法插入\n");
        return 0;
    }

    // 合法的插入位置是 1 到 length+1（也就是允许插在最末尾）
    if (pos < 1 || pos > L->length + 1) 
    {
        printf("插入位置错误\n");
        return 0;
    }

    // 只有当插入位置在已有元素中间时，才需要把后面的元素往后挪
    if (pos <= L->length) 
    {
        // 核心易错点：必须从后往前挪，否则前面的数据会覆盖后面的数据
        for (int i = L->length - 1; i >= pos - 1; i--)
        {
            L->data[i + 1] = L->data[i];
        }
    }
    
    // 腾出空位后，执行赋值和长度增加（修复了之前的尾插 Bug）
    L->data[pos - 1] = e; 
    L->length++;
    
    return 1;
}


/*
 * 5. 删除元素（时间复杂度 O(n)）
 * @param e: 指针传参，用于把被删除的元素“带回”给主函数
 */
int deleteElem(SeqList* L, int pos, ElemType* e)
{
    if (L->length == 0)
    {
        printf("空表，无法删除\n");
        return 0;
    }

    // 删除位置只能是 1 到 length
    if (pos < 1 || pos > L->length) 
    {
        printf("删除位置错误\n");
        return 0;
    }

    // 将要被覆盖抹杀的数据，提前存入 e 指向的地址中保全下来
    *e = L->data[pos - 1]; 

    // 如果删除的不是最后一个元素，则需要把后面的数据整体往前挪一位
    if (pos < L->length) 
    {
        // 从前往后挪，依次覆盖
        for (int i = pos; i < L->length; i++)
        {
            L->data[i - 1] = L->data[i];
        }
    }
    L->length--; // 删除成功，表长度-1
    return 1;
}


/*
 * 6. 查找元素
 * 返回该元素在表中的逻辑位置（从 1 开始），找不到返回 0
 */
int findElem(SeqList* L, ElemType e)
{
    for (int i = 0; i < L->length; i++)
    {
        if (L->data[i] == e)
        {
            return i + 1; // 数组下标 i 对应的现实位置是 i + 1
        }
    }
    return 0;
}


int main()
{
    // 1.声明并初始化
    SeqList *list1 = initList();
    
    printf("目前长度占用：%d\n", (*list1).length);
    // 【避坑笔记】因为 data 是指针，所以 sizeof 算出来的只是指针本身的大小（8字节），不是整个仓库的真实大小
    printf("目前 data 指针自身内存占用：%zu 字节\n\n", sizeof((*list1).data));

    // 2.测试添加
    printf("--- 开始添加元素 ---\n");
    appendElem(list1, 52);
    appendElem(list1, 11);
    appendElem(list1, 22);
    listElem(list1);

    // 3.测试插入
    printf("--- 在第2个位置插入44 ---\n");
    insertElem(list1, 2, 44);
    listElem(list1);

    // 4.测试删除
    printf("--- 删除第2个位置的元素 ---\n");
    ElemType delData; // 定义一个普通变量用来接住被删掉的值
    deleteElem(list1, 2, &delData); // 传入该变量的地址
    printf("被删除的数据为：%d\n", delData);
    listElem(list1);

    // 5.测试查找
    printf("--- 查找数据 22 的位置 ---\n");
    int weizhi = findElem(list1, 22);
    if(weizhi != 0)
        printf("查找的数据位置是：第 %d 个\n", weizhi);
    else
        printf("表中没有该元素\n");

    free(list1->data); // 先释放里面大仓库的内存
    free(list1);       // 再释放结构体本身的内存
    
    return 0;
}
